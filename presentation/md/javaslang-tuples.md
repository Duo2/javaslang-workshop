## Javaslang Tuple

    private Tuple2<String, Integer> filesInPath(
            String basePath, 
            String fileExtension) {
        Integer fileCount = findAllIn(basePath, fileExtension).length();
        return Tuple.of(basePath, fileCount);
    }
    
    String count = filesInPath("/", "*.txt")._2;
